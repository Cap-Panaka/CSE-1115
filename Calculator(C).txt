#include <stdio.h>
 float summation(float x,float y);
    float subtraction(float x,float y);
    float multiplication(float x,float y);
    float division(float x,float y);

int main(){
    float x,y,add,sub,multi,divi;
    printf("-------Calculator.exe---------\n");
    printf("Enter the first number(x): ");
    scanf("%f",&x);
    printf("Enter the second number(y): ");
    scanf("%f",&y);
    
    add=summation(x,y);
    printf("Summation(x+y): %f\n",add);
    
    sub=subtraction(x,y);
    printf("Subtraction(x-y): %f\n",sub);
    
    multi=multiplication(x,y);
    printf("Multiplication(x*y): %f\n",multi);
    
    divi=division(x,y);
    printf("Division(x/y): %f\n",divi);
    printf("---------------x--------------");
    return 0;
}
    float summation(float x,float y){
        return x+y;
    }
    float subtraction(float x,float y){
        return x-y;
    }
    float multiplication(float x,float y){
        return x*y;
    }
    float division(float x,float y){
        return x/y;
    }
